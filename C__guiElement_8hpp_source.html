276 fge::GuiElementContext context{};

278 context._mouseGuiPosition = this->g_target->mapPixelToCoords(context._mousePosition, this->g_target->getDefaultView());
279 context._handler = this;
280
281 this->_onGuiVerify.call(evt, sf::Event::EventType::MouseWheelScrolled, context);

287 context._prioritizedElement->_onGuiMouseWheelScrolled.call(evt, arg, context);
288 context._recursive = true;
289 auto* element = context._prioritizedElement;
290 context._prioritizedElement = nullptr;
291 element->onGuiVerify(evt, sf::Event::EventType::MouseWheelScrolled, context);
292 if (context._prioritizedElement != nullptr)
293 {
294 context._prioritizedElement->_onGuiMouseWheelScrolled.call(evt, arg, context);

469class GuiElementArray : public fge::GuiElement
470{
471public:
472 GuiElementArray() = default;
473 explicit GuiElementArray(fge::GuiElement::Priority priority) :
474 fge::GuiElement(priority)
475 {}
476 ~GuiElementArray() override = default;
477
478 [[nodiscard]] bool isRecursive() const override
479 {
480 return true;
481 }
482
483 void onGuiVerify(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) override
484 {
485 if (context._recursive)
486 {
487 this->verifyRecursively(evt, evtType, context);
488 }
489 else
490 {
491 if ( this->verifyPriority(context._prioritizedElement) )
492 {
493 context._prioritizedElement = this;
494 }
495 }
496 }
497
498 fge::Tunnel<fge::GuiElement> _elements;
499
500protected:
501 void verifyRecursively(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) const
502 {
503 fge::GuiElementContext context2{};
504 context2._mouseGuiPosition = context._mouseGuiPosition;
505 context2._mousePosition = context._mousePosition;
506 context2._handler = context._handler;
507
508 for (std::size_t i=0; i<this->_elements.getGatesSize(); ++i)
509 {
510 context2._index = i;
511 this->_elements[i]->onGuiVerify(evt, evtType, context2);
512 }
513
514 if (context2._prioritizedElement != nullptr)
515 {
516 if (context2._prioritizedElement->isRecursive())
517 {
518 context2._recursive = true;
519 context2._prioritizedElement = nullptr;
520 context2._prioritizedElement->onGuiVerify(evt, evtType, context2);
521 }
522 }
523
524 context._prioritizedElement = context2._prioritizedElement;
525 context._index = context2._index;
526 }
527};
