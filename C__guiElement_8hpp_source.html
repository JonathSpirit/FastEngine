264 void onMouseWheelScrolled(const fge::Event& evt, const sf::Event::MouseWheelScrollEvent& arg)
265 {
266 fge::GuiElementContext context{};
267 context._mousePosition = {arg.x, arg.y};
268 context._mouseGuiPosition = this->g_target->mapPixelToCoords(context._mousePosition, this->g_target->getDefaultView());
269 context._handler = this;
270
271 this->_onGuiVerify.call(evt, sf::Event::EventType::MouseWheelScrolled, context);
272
273 if (context._prioritizedElement != nullptr)
274 {
275 if (context._prioritizedElement->isRecursive())
276 {
277 context._prioritizedElement->_onGuiMouseWheelScrolled.call(evt, arg, context);
278 context._recursive = true;
279 auto* element = context._prioritizedElement;
280 context._prioritizedElement = nullptr;
281 element->onGuiVerify(evt, sf::Event::EventType::MouseWheelScrolled, context);
282 if (context._prioritizedElement != nullptr)
283 {
284 context._prioritizedElement->_onGuiMouseWheelScrolled.call(evt, arg, context);
285 }
286 }
287 else
288 {
289 context._prioritizedElement->_onGuiMouseWheelScrolled.call(evt, arg, context);

393class GuiElementRectangle : public fge::GuiElement

395public:

398 fge::GuiElement(priority),
399 g_rect(rect)
400 {}
401 explicit GuiElementRectangle(fge::GuiElement::Priority priority) :
402 fge::GuiElement(priority)
403 {}
404 ~GuiElementRectangle() override = default;
405
406 void onGuiVerify(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) override
407 {
408 if ( this->verifyPriority(context._prioritizedElement) )
409 {
410 sf::FloatRect rect{this->g_rect.getPosition(), {this->g_rect.width*this->_g_scale.x, this->g_rect.height*this->_g_scale.y}};
411 if ( rect.contains(context._mouseGuiPosition) )
412 {
413 context._prioritizedElement = this;
414 }
415 }
416 }

459class GuiElementArray : public fge::GuiElement
460{
461public:
462 GuiElementArray() = default;
463 explicit GuiElementArray(fge::GuiElement::Priority priority) :
464 fge::GuiElement(priority)
465 {}
466 ~GuiElementArray() override = default;
467
468 [[nodiscard]] bool isRecursive() const override
469 {
470 return true;
471 }
472
473 void onGuiVerify(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) override
474 {
475 if (context._recursive)
476 {
477 this->verifyRecursively(evt, evtType, context);
478 }
479 else
480 {
481 if ( this->verifyPriority(context._prioritizedElement) )
482 {
483 context._prioritizedElement = this;
484 }
485 }
486 }
487
488 fge::Tunnel<fge::GuiElement> _elements;
489
490protected:
491 void verifyRecursively(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) const
492 {
493 fge::GuiElementContext context2{};
494 context2._mouseGuiPosition = context._mouseGuiPosition;
495 context2._mousePosition = context._mousePosition;
496 context2._handler = context._handler;
497
498 for (std::size_t i=0; i<this->_elements.getGatesSize(); ++i)
499 {
500 context2._index = i;
501 this->_elements[i]->onGuiVerify(evt, evtType, context2);
502 }
503
504 if (context2._prioritizedElement != nullptr)
505 {
506 if (context2._prioritizedElement->isRecursive())
507 {
508 context2._recursive = true;
509 context2._prioritizedElement = nullptr;
510 context2._prioritizedElement->onGuiVerify(evt, evtType, context2);
511 }
512 }
513
514 context._prioritizedElement = context2._prioritizedElement;
515 context._index = context2._index;
516 }
517};
