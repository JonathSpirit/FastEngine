52struct DynamicSize
53{
54 enum class SizeModes
55 {
56 SIZE_FIXED,
57 SIZE_AUTO,
58 SIZE_DEFAULT = SIZE_AUTO
59 };
60
61 sf::Vector2f _fixedSize{0.0f,0.0f};
62 sf::Vector2<fge::DynamicSize::SizeModes> _sizeMode{fge::DynamicSize::SizeModes::SIZE_DEFAULT,
63 fge::DynamicSize::SizeModes::SIZE_DEFAULT};
64 sf::Vector2f _offset{0.0f,0.0f};
65
66 [[nodiscard]] inline sf::Vector2f getSize(const sf::Vector2f& position, const sf::Vector2f& targetSize) const
67 {
68 sf::Vector2f size{};
69
70 switch (this->_sizeMode.x)
71 {
72 case SizeModes::SIZE_FIXED:
73 size.x = this->_fixedSize.x;
74 break;
75 case SizeModes::SIZE_AUTO:
76 size.x = (targetSize.x-position.x) + this->_offset.x;
77 break;
78 default:
79 break;
80 }
81
82 switch (this->_sizeMode.y)
83 {
84 case SizeModes::SIZE_FIXED:
85 size.y = this->_fixedSize.y;
86 break;
87 case SizeModes::SIZE_AUTO:
88 size.y = (targetSize.y-position.y) + this->_offset.y;
89 break;
90 default:
91 break;
92 }
93
94 if (size.x < 0.0f)
95 {
96 size.x = 0.0f;
97 }
98 if (size.y < 0.0f)
99 {
100 size.y = 0.0f;
101 }
102
103 return size;
104 }
105};

479class GuiElementArray : public fge::GuiElementRecursive
480{
481public:
482 GuiElementArray() = default;
483 explicit GuiElementArray(fge::GuiElement::Priority priority) :
484 fge::GuiElement(priority)
485 {}
486 ~GuiElementArray() override = default;
487
488 void onGuiVerify(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) override
489 {
490 if (context._recursive)
491 {
492 this->verifyRecursively(evt, evtType, context);
493 }
494 else
495 {
496 if ( this->verifyPriority(context._prioritizedElement) )
497 {
498 context._prioritizedElement = this;
499 }
500 }
501 }
502
503 fge::Tunnel<fge::GuiElement> _elements;
504
505protected:
506 void verifyRecursively(const fge::Event& evt, sf::Event::EventType evtType, fge::GuiElementContext& context) const
507 {
508 fge::GuiElementContext context2{};
509 context2._mouseGuiPosition = context._mouseGuiPosition;
510 context2._mousePosition = context._mousePosition;
511 context2._handler = context._handler;
512
513 for (std::size_t i=0; i<this->_elements.getGatesSize(); ++i)
514 {
515 context2._index = i;
516 this->_elements[i]->onGuiVerify(evt, evtType, context2);
517 }
518
519 if (context2._prioritizedElement != nullptr)
520 {
521 if (context2._prioritizedElement->isRecursive())
522 {
523 context2._recursive = true;
524 context2._prioritizedElement->onGuiVerify(evt, evtType, context2);
525 }
526 }
527
528 context._prioritizedElement = context2._prioritizedElement;
529 context._index = context2._index;
530 }
