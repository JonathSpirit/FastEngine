102template<class... Types>
103fge::CallbackBase<Types...>* CallbackHandler<Types...>::add(CalleePtr&& callback, fge::Subscriber* subscriber)
104{
105 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
106
107 this->attach(subscriber);
108
109 //Search for a "mark for deletion" callee and replace it with the new one
110 for (auto& callee: this->g_callees)
111 {
112 if (callee._f == nullptr)
113 {
114 callee._f = std::move(callback);
115 callee._subscriber = subscriber;
116 return callee._f.get();
117 }
118 }
119
120 //Emplace back the new callee if there is no "mark for deletion"
121 this->g_callees.emplace_back(std::move(callback), subscriber);
122 return this->g_callees.back()._f.get();
123}

151template<class... Types>
152void CallbackHandler<Types...>::delPtr(void* ptr)
153{
154 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
155 for (auto& callee: this->g_callees)
156 {
157 if (callee._f == nullptr)
158 {
159 continue;
160 }
161
162 if (callee._f->check(ptr))
163 {
164 this->detachOnce(callee._subscriber);
165 callee._f = nullptr;
166 }
167 }
168}
169template<class... Types>
170void CallbackHandler<Types...>::delSub(fge::Subscriber* subscriber)
171{
172 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
173 for (auto& callee: this->g_callees)
174 {
175 if (callee._f == nullptr)
176 {
177 continue;
178 }
179
180 if (callee._subscriber == subscriber)
181 {
182 if (this->detachOnce(callee._subscriber) == 0)
183 { //At this point, all subscribers are detached so we can end the loop
184 callee._f = nullptr;
185 break;
186 }
187 callee._f = nullptr;
188 }
189 }

192void CallbackHandler<Types...>::del(fge::CallbackBase<Types...>* callback)
193{

195 for (auto& callee: this->g_callees)
196 {

210template<class... Types>
211void CallbackHandler<Types...>::call(Types... args)
212{
213 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
214
215 std::size_t eraseCount = 0;
216 for (std::size_t i = 0; i < this->g_callees.size(); ++i)
217 {
218 if (this->g_callees[i]._f == nullptr)
219 {
220 ++eraseCount;
221 continue;
222 }
223
224 if (eraseCount > 0)
225 { //We found a valid callee, we can erase the previous ones
226 this->g_callees.erase(this->g_callees.begin() + i - eraseCount, this->g_callees.begin() + i);
227 i -= eraseCount;
228 eraseCount = 0;
229 }
230
231 this->g_callees[i]._f->call(std::forward<Types>(args)...);
232
233 //While calling, the callee can remove itself or others
234 //Every erased callee is replaced by nullptr (mark for deletion)
235 //In order to avoid calling somthing that is not valid anymore
236 }
237}

239template<class... Types>
240void CallbackHandler<Types...>::hook(fge::CallbackHandler<Types...>& handler, fge::Subscriber* subscriber)
241{
242 if (this == &handler)
243 { //Can't hook itself
244 return;
245 }
246
247 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
248 this->add(new fge::CallbackLambda<Types...>(
249 [&handler](Types... args) { handler.call(std::forward<Types>(args)...); }),
250 subscriber);
251}

253template<class... Types>
254void CallbackHandler<Types...>::onDetach(fge::Subscriber* subscriber)
255{
256 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
257 for (auto& callee: this->g_callees)
258 {
259 if (callee._f == nullptr)
260 {
261 continue;
262 }
263
264 if (callee._subscriber == subscriber)
265 {
266 callee._f = nullptr;
267 }
