99template<class... Types>
100void CallbackHandler<Types...>::add(fge::CallbackFunctorBase<Types...>* callback, fge::Subscriber* subscriber)
101{
102 std::lock_guard<std::recursive_mutex> lck(this->g_mutex);
103 this->attach(subscriber);
104 this->g_callees.push_front({typename fge::CallbackHandler<Types...>::CalleePtr(callback), subscriber});
105}
106template<class... Types>
107void CallbackHandler<Types...>::delPtr(void* ptr)
108{
109 std::lock_guard<std::recursive_mutex> lck(this->g_mutex);
110 typename fge::CallbackHandler<Types...>::CalleeList::iterator prev = this->g_callees.before_begin();
111 for (typename fge::CallbackHandler<Types...>::CalleeList::iterator it = this->g_callees.begin();
112 it != this->g_callees.end(); ++it)
113 {
114 if ((*it)._f->check(ptr))
115 {
116 if (this->detachOnce((*it)._subscriber) == 0)
117 {
118 this->g_callees.erase_after(prev);
119 break;
120 }
121 this->g_callees.erase_after(prev);
122 it = prev;
123 }
124 else
125 {
126 prev = it;
127 }
128 }
129}

155template<class... Types>
156void CallbackHandler<Types...>::call(Types... args)
157{
158 std::lock_guard<std::recursive_mutex> lck(this->g_mutex);
159 auto itCalleeNext = this->g_callees.begin();
160 for (auto itCallee = this->g_callees.begin(); itCallee != this->g_callees.end(); itCallee = itCalleeNext)
161 {
162 ++itCalleeNext;
163 itCallee->_f->call(args...);
164 }
