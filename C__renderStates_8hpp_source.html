174class RenderResourceTextures
175{
176public:
185 enum class PtrTypes
186 {
187 TEXTURE,
188 TEXTURE_IMAGE
189 };
190
191 template<class...>
192 static constexpr std::false_type always_false{};
193
194 constexpr RenderResourceTextures() = default;
195 template<class TTexture>
196 constexpr RenderResourceTextures(TTexture const* textures, uint32_t count) :
197 g_textures(textures),
198 g_count(textures == nullptr ? 0 : count)
199 {
200 if constexpr (std::is_same_v<TTexture, fge::TextureType> || std::is_same_v<TTexture, std::nullptr_t>)
201 {
202 this->g_ptrType = PtrTypes::TEXTURE_IMAGE;
203 }
204 else if constexpr (std::is_same_v<TTexture, fge::Texture>)
205 {
206 this->g_ptrType = PtrTypes::TEXTURE;
207 }
208 else
209 {
210 static_assert(always_false<TTexture>, "Can't set this type of texture pointer !");
211 }
212 }
213
214 template<class TTexture>
215 constexpr void set(TTexture const* textures, uint32_t count)
216 {
217 this->g_textures = textures;
218 this->g_count = textures == nullptr ? 0 : count;
219
220 if constexpr (std::is_same_v<TTexture, fge::TextureType> || std::is_same_v<TTexture, std::nullptr_t>)
221 {
222 this->g_ptrType = PtrTypes::TEXTURE_IMAGE;
223 }
224 else if constexpr (std::is_same_v<TTexture, fge::Texture>)
225 {
226 this->g_ptrType = PtrTypes::TEXTURE;
227 }
228 else
229 {
230 static_assert(always_false<TTexture>, "Can't set this type of texture pointer !");
231 }
232 }
233 [[nodiscard]] constexpr uint32_t getCount() const { return this->g_count; }
234 [[nodiscard]] constexpr PtrTypes getPointerType() const { return this->g_ptrType; }
235
236 template<PtrTypes TTextureType>
237 constexpr auto const* get(uint32_t index) const
238 {
239 if constexpr (TTextureType == PtrTypes::TEXTURE_IMAGE)
240 {
241 return static_cast<fge::TextureType const*>(this->g_textures) + index;
242 }
243 else
244 {
245 return static_cast<fge::Texture const*>(this->g_textures) + index;
246 }
247 }
248
249 template<PtrTypes TTextureType>
250 [[nodiscard]] constexpr fge::TextureType const* getTextureImage(uint32_t index) const
251 {
252 if constexpr (TTextureType == PtrTypes::TEXTURE_IMAGE)
253 {
254 return static_cast<fge::TextureType const*>(this->g_textures) + index;
255 }
256 else
257 {
258#ifndef FGE_DEF_SERVER
259 return static_cast<fge::Texture const*>(this->g_textures)[index].getSharedTexture().get();
260#else
261 return nullptr;
262#endif
263 }
264 }
265
266private:
267 void const* g_textures{nullptr};
268 PtrTypes g_ptrType{PtrTypes::TEXTURE_IMAGE};
269 uint32_t g_count{0};
