60template<class TReturn, class... Types>
61template<typename TLambda>
62CallbackLambda<TReturn, Types...>::CallbackLambda(TLambda const& lambda) :
63 g_lambda(new TLambda(lambda))
64{
65 this->g_executeLambda = [](void* lambdaPtr, [[maybe_unused]] Types... arguments) {
66 if constexpr (std::is_invocable_v<TLambda, Types...>)
67 {
68 return (*reinterpret_cast<TLambda*>(lambdaPtr))(arguments...);
69 }
70 else
71 {
72 return (*reinterpret_cast<TLambda*>(lambdaPtr))();
73 }
74 };
75 this->g_deleteLambda = [](void* lambdaPtr) { delete reinterpret_cast<TLambda*>(lambdaPtr); };
76}
77template<class TReturn, class... Types>
78CallbackLambda<TReturn, Types...>::~CallbackLambda()
79{
80 (*this->g_deleteLambda)(this->g_lambda);
81}

83template<class TReturn, class... Types>
84TReturn CallbackLambda<TReturn, Types...>::call(Types... args)
85{
86 return (*this->g_executeLambda)(this->g_lambda, args...);
87}

107template<class... Types>
108fge::CallbackBase<void, Types...>* CallbackHandler<Types...>::add(CalleePtr&& callback, fge::Subscriber* subscriber)
109{
110 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
111
112 this->attach(subscriber);
113
114 //Search for a "mark for deletion" callee and replace it with the new one
115 for (auto& callee: this->g_callees)
116 {
117 if (callee._markedForDeletion)
118 {
119 //TODO: _markedForDeletion has been added in order to avoid undefined behavior when a callback remove itself
120 // while being called. But here we do the same mistake, we replace a callee that could be called at the same time.
121 // We need a better solution.
122 callee._f = std::move(callback);
123 callee._subscriber = subscriber;
124 callee._markedForDeletion = false;
125 return callee._f.get();
126 }
127 }
128
129 //Emplace back the new callee if there is no "mark for deletion"
130 this->g_callees.emplace_back(std::move(callback), subscriber);
131 return this->g_callees.back()._f.get();
132}

160template<class... Types>
161void CallbackHandler<Types...>::delPtr(void* ptr)
162{
163 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
164 for (auto& callee: this->g_callees)
165 {
166 if (callee._markedForDeletion)
167 {
168 continue;
169 }
170
171 if (callee._f->check(ptr))
172 {
173 this->detachOnce(callee._subscriber);
174 callee._markedForDeletion = true;
175 }
176 }

178template<class... Types>
179void CallbackHandler<Types...>::delSub(fge::Subscriber* subscriber)
180{
181 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
182 for (auto& callee: this->g_callees)
183 {
184 if (callee._markedForDeletion)
185 {
186 continue;
187 }
188
189 if (callee._subscriber == subscriber)
190 {
191 if (this->detachOnce(callee._subscriber) == 0)
192 { //At this point, all subscribers are detached so we can end the loop
193 callee._markedForDeletion = true;
194 break;
195 }
196 callee._markedForDeletion = true;
197 }
198 }

219template<class... Types>
220void CallbackHandler<Types...>::call(Types... args)
221{
222 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
223
224 std::size_t eraseCount = 0;
225 for (std::size_t i = 0; i < this->g_callees.size(); ++i)
226 {
227 if (this->g_callees[i]._markedForDeletion)
228 {
229 ++eraseCount;
230 continue;
231 }
232
233 if (eraseCount > 0)
234 { //We found a valid callee, we can erase the previous ones
235 this->g_callees.erase(this->g_callees.begin() + i - eraseCount, this->g_callees.begin() + i);
236 i -= eraseCount;
237 eraseCount = 0;
238 }
239
240 this->g_callees[i]._f->call(std::forward<Types>(args)...);
241
242 //While calling, the callee can remove itself or others
243 //Every erased callee is flagged (mark for deletion)
244 //In order to avoid calling something that is not valid anymore
245 }

248template<class... Types>
249void CallbackHandler<Types...>::hook(fge::CallbackHandler<Types...>& handler, fge::Subscriber* subscriber)
250{
251 if (this == &handler)
252 { //Can't hook itself
253 return;
254 }
255
256 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
257 this->add(new fge::CallbackLambda<Types...>(
258 [&handler](Types... args) { handler.call(std::forward<Types>(args)...); }),
259 subscriber);
260}

292 fge::Subscriber* subscriber)
293{
294 std::scoped_lock<std::recursive_mutex> const lck(this->g_mutex);
295
296 if (subscriber != this->g_callee._subscriber)
297 {
298 this->detachOnce(this->g_callee._subscriber);
299 this->attach(subscriber);
300 }
301
302 this->g_callee._f = std::move(callback);
303 this->g_callee._subscriber = subscriber;
