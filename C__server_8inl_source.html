61template<typename Tpacket>
62void ServerUdp::serverThreadReception()
63{
64 fge::net::Identity idReceive;
65 Tpacket pckReceive;
66 std::size_t pushingIndex = 0;
67
68 while (this->g_running)
69 {
70 if (this->g_socket.select(true, 500) == fge::net::Socket::ERR_NOERROR)
71 {
72 if (this->g_socket.receiveFrom(pckReceive, idReceive._ip, idReceive._port) == fge::net::Socket::ERR_NOERROR)
73 {
74 std::lock_guard<std::mutex> lck(this->g_mutexServer);
75 if (this->g_flux.empty())
76 {
77 this->g_defaultFlux.pushPacket(
78 std::make_shared<fge::net::FluxPacket>(std::move(pckReceive), idReceive));
79 continue;
80 }
81
82 pushingIndex = (pushingIndex + 1) % this->g_flux.size();
83 //Try to push packet in a flux
84 for (std::size_t i = 0; i < this->g_flux.size(); ++i)
85 {
86 pushingIndex = (pushingIndex + 1) % this->g_flux.size();
87 if (this->g_flux[pushingIndex]->pushPacket(
88 std::make_shared<fge::net::FluxPacket>(std::move(pckReceive), idReceive, pushingIndex)))
89 {
90 //Packet is correctly pushed
91 break;
92 }
93 }
94 //If every flux is busy, the new packet is dismissed
95 }
96 }
97 }
98}

152template<typename Tpacket>
153void ServerClientSideUdp::serverThreadReception()
154{
155 Tpacket pckReceive;
156
157 while (this->g_running)
158 {
159 if (this->g_socket.select(true, 500) == fge::net::Socket::ERR_NOERROR)
160 {
161 if (this->g_socket.receive(pckReceive) == fge::net::Socket::ERR_NOERROR)
162 {
163 this->pushPacket(std::make_shared<fge::net::FluxPacket>(std::move(pckReceive), this->g_clientIdentity));
164 this->g_cvReceiveNotifier.notify_all();
165 }
166 }
167 }
